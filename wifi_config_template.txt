# WiFi Configuration Template for Raspberry Pi
# Save this as wpa_supplicant.conf on the boot partition of your SD card

update_config=1

# Your current WiFi (for setup and testing)
# Higher priority = preferred network

    ssid="YourCurrentWiFiName"
    psk="YourCurrentWiFiPassword"
    priority=10
}

    ssid="YourOtherHouseWiFiName"
    psk="YourOtherHouseWiFiPassword"
